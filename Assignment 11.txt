int getLcm(int a,int b)
{
	int lcm;
	if(a>b)
	lcm=a;
	else
	lcm=b;
	while(1)
	{
		if(lcm%a==0&&lcm%b==0)
		{
			//cout<<"LCM OF"<<a<<"and"<<b<<"is     "<<lcm;
			break;
		}
		lcm++;
	}
	return lcm;
}

        }

    }

        scanf("%d", & n);  
        for (i = 0; i < n; i++) {  
            for (c = 0; c <= (n - i - 2); c++) printf(" ");  
            for (c = 0; c <= i; c++) printf("%ld ", factorial(i) / (factorial(c) * factorial(i - c)));  
            printf("\n");  
        }  
        return 0;  
    }  
    long factorial(int n) {  
        int c;  
        long result = 1;  
        for (c = 1; c <= n; c++) result = result * c;  
        return result;  
    }
